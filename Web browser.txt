WebView webView;
webView = (WebView) findViewById(R.id.webView);

        webView.loadUrl("file:///android_asset/ananta prem.html");
        webView.setWebViewClient(new WebViewClient());

            webView.goBack();

            super.onBackPressed();
        }
    }
